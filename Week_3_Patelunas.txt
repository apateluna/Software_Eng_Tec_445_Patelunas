//tic-tak-toe board
//Demonstrates MULITDIMENSIONAL ARRAYS


#include<iostream>
#include<string>
#include<vector>

using  namespace::std;
using  std::cout;
using  std::cin;

int main ()
{
const int ROWS=5;
const int COLUMNS =5;
char board[ROWS][COLUMNS]={{'O','X','O'},
                           {' ','X','X'},
                           {'X','O','O'} };
cout<<" Here is the tic-tak-toe board:\n";
for (int i = 0; i < ROWS; ++i)
{
for(int j=0; j<COLUMNS; ++j)
{ 
cout << board [i][j];
}
cout<<endl;
}
cout<<"\n'X' moves to the empty location.\n\n";
board[1][0] = 'X';
cout<<"Now the tic-tak-toe board is:\n";
for(int i=0; i< ROWS;++i)
{
for(int j=0; j<COLUMNS; ++j)
{
cout<<board[i][j];
}
cout<<endl;
}
cout<<"\n'X' WINNA!!!!";
return 0;
}
